40  	final int __NAME__SORT = 1;
41  
42  	@GetMapping("/AltaAutor")
43  	public String altaAutor(Model model) {
44          List<Autor> listadoAutores = autorDAO.findAll();
45  		// Ordenamos la lista por nombre
46  		if (__NAME__SORT == 1) {
47  			Collections.sort(listadoAutores, new TitleComparator());
48  		}
49  
50          model.addAttribute("titulo", "Lista de Autores");
51          model.addAttribute("autor", new Autor());
52          model.addAttribute("listautores", listadoAutores);
53  
54  		log.info(autorDAO.findAll().toString());
55  		return "autor/AltaAutor";
56  	}

58  	@PostMapping("/AltaAutor")
59  	public String autorSubmit(@ModelAttribute Autor autor, Model model, RedirectAttributes attribute) {
60          this.autor = autor;
61  		model.addAttribute("autor", autor);
62  		Autor savedAutor = autorDAO.save(autor);
63  		log.info("Saved autor: " + savedAutor);
64  		// titulo.addAutor(autor);
65  		attribute.addFlashAttribute("success", "El autor se ha creado con éxito");
66  		return "redirect:/AltaAutor";
67  	}

69  	@GetMapping("/AltaAutor/edit/{id}")
70  	public String formularioEditarAutor(@PathVariable("id") long id, Model model) {

72  		model.addAttribute("autor", autor);

76  	@PostMapping("/AltaAutor/update/{id}")

78  		autorDAO.save(autor);
79  		attribute.addFlashAttribute("info", "El autor ha sido modificado");
80  		return "redirect:/AltaAutor";
81  	}
82  
83  	@GetMapping("/AltaAutor/delete/{id}")
84  	public String deleteAutor(@PathVariable("id") long id, RedirectAttributes attribute) {
85  		suprimirAsociacionesAutorLibro(autorDAO.findById(id));
86  		autorDAO.deleteById(id);
87  		attribute.addFlashAttribute("warning", "El autor se ha eliminado");
88  		return "redirect:/AltaAutor";
89  	}
90  
91  	@PreRemove
92  	private void suprimirAsociacionesAutorLibro(Optional<Autor> au) {
93  		// Método anterior a eliminación de autor por DAO (=PreRemove)
94  		// Revisa cada título asociado con autor y se elimina el vínculo, manteniendo el título
95  		Autor a = au.get();
96  		for (Titulo t : a.getTitulos()){
97  			t.getAutores().remove(a);
98  		}
99  	}
